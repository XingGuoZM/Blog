
### 函数合成  
```
function func1(){
  console.log(1)
}
function func2(){
  console.log(2)
}
function func3(){
  console.log(3)
}

function func4(){
  console.log(4)
}
function func5(){
  console.log(5)
}
// 方法一：
function compose1(){
  const funcs=[...arguments][0]
  return funcs.reduce((a, b) => (...args) => a(b(...args)))()
}
//方法二：
function compose2(){
  const funcs=[...arguments][0]
  let prev
  for(let i=funcs.length-1;i>=0;i--){
    prev=funcs[i](prev)
  }
  return prev || function(){}
}
const funcs=[func1,func2,func3,func4,func5]
compose1('方法一输出：',funcs)
compose2('方法二输出：',funcs)
```
